import math
directory = "./Tracks"
csv_files = glob.glob(f"{directory}/*.csv") 

Angles_XZ = []
Angles_YZ = []
count = 0

for file_ in csv_files:
    with open(file_, mode ='r') as file:    
       csvFile = csv.DictReader(file)
       for lines in csvFile:
            if(int(lines['trType'])) == 1:
                angle_xz = math.atan(float(lines['slopeXZ']))*(180/math.pi)
                Angles_XZ.append(angle_xz)
                angle_yz = math.atan(float(lines['slopeYZ']))*(180/math.pi)
                Angles_YZ.append(angle_yz)
